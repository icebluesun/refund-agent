2. UNIT TESTING THE GRAPH (Not Just Functions)

Why test the graph? Because the flow IS the logic.

Using LangGraph's testing utilities:

from langgraph.graph import StateGraph
from langgraph.testing import assert_graph_transitions

def test_purchase_workflow():
    """Verify correct node transitions for valid purchase"""
    initial_state = {
        "customer_id": "CUST-001",
        "customer_tier": "PLATINUM",
        "item_id": "ITEM-001",
        "requested_quantity": 1,
        "price_per_unit": 149.99
    }
    
    # Assert: entry -> verify_limits -> commit_ledger -> END
    assert_graph_transitions(
        purchase_app,
        initial_state,
        expected_path=["verify_limits", "commit_ledger", END]
    )

def test_rejected_purchase():
    """Verify correct transitions when cap exceeded"""
    # Customer already owns 5 units
    initial_state = {
        "customer_id": "CUST-001",
        "item_id": "ITEM-001",
        "requested_quantity": 1
    }
    
    # Assert: entry -> verify_limits -> terminal_reject -> END
    assert_graph_transitions(
        purchase_app,
        initial_state,
        expected_path=["verify_limits", "terminal_reject", END]
    )

def test_defect_approval_flow():

3. INTEGRATION TESTING

Test the full user journey:

def test_complete_refund_flow():
    """End-to-end: purchase -> refund -> approve"""
    # Step 1: Purchase
    purchase_result = execute_purchase(
        "CUST-001", "PLATINUM", "ITEM-001", 1, 149.99
    )
    assert purchase_result["ledger_committed"] == True
    
    # Step 2: Get the new unit ID
    units = db.get_customer_units("CUST-001", "ITEM-001")
    new_unit = [u for u in units if u["status"] == "ACTIVE"][-1]
    
    # Step 3: Request refund
    req = db.create_refund_request(
        "CUST-001", "ITEM-001", "arrived damaged", 
        [new_unit["unit_id"]], 149.99
    )
    
    # Step 4: Evaluate
    result = execute_dispute_evaluation(req, {
        "customer_id": "CUST-001",
        "customer_tier": "PLATINUM",
        "item_id": "ITEM-001",
        "total_cost": 149.99
    })
    
    # Step 5: Verify
    assert result["decision"] == "Approved"
    
    # Step 6: Verify unit state changed
    updated = db.get_unit_by_id(new_unit["unit_id"])
    assert updated["status"] == "REFUNDED"
